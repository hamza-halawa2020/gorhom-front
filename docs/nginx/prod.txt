# path: /etc/nginx/sites-available/hpsaviation.com

# Production server
server {
    root /var/www/hps/dist/app/browser;
    index index.html;
    server_name hpsaviation.com www.hpsaviation.com;

    # Plausible Analytics Script Proxy
    # This serves the Plausible script from your domain to bypass ad-blockers
    location /js/script.js {
        # Proxy to Plausible's CDN
        proxy_pass https://plausible.io/js/script.js;
        
        # Set proper headers for Plausible
        proxy_set_header Host plausible.io;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        
        # Cache the script for 1 hour
        proxy_cache_valid 200 1h;
        proxy_cache_key $scheme$proxy_host$request_uri;
        
        # Add CORS headers if needed
        add_header Access-Control-Allow-Origin *;
        add_header Access-Control-Allow-Methods "GET, POST, OPTIONS";
        add_header Access-Control-Allow-Headers "Content-Type, Authorization";
        
        # Handle preflight requests
        if ($request_method = 'OPTIONS') {
            add_header Access-Control-Allow-Origin *;
            add_header Access-Control-Allow-Methods "GET, POST, OPTIONS";
            add_header Access-Control-Allow-Headers "Content-Type, Authorization";
            add_header Content-Length 0;
            add_header Content-Type text/plain;
            return 204;
        }
    }

    # Plausible Analytics API Proxy
    # This proxies analytics events to Plausible's API
    location /api/event {
        # Proxy to Plausible's API
        proxy_pass https://plausible.io/api/event;
        
        # Set proper headers for Plausible API
        proxy_set_header Host plausible.io;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header Content-Type application/json;
        
        # Handle CORS for API requests
        add_header Access-Control-Allow-Origin *;
        add_header Access-Control-Allow-Methods "POST, OPTIONS";
        add_header Access-Control-Allow-Headers "Content-Type, Authorization";
        
        # Handle preflight requests
        if ($request_method = 'OPTIONS') {
            add_header Access-Control-Allow-Origin *;
            add_header Access-Control-Allow-Methods "POST, OPTIONS";
            add_header Access-Control-Allow-Headers "Content-Type, Authorization";
            add_header Content-Length 0;
            add_header Content-Type text/plain;
            return 204;
        }
        
        # Only allow POST requests for the API
        limit_except POST OPTIONS {
            deny all;
        }
    }

    # PHP files from project root (scripts directory)
    location ~ ^/scripts/.*\.php$ {
        root /var/www/hps;
        include snippets/fastcgi-php.conf;
        fastcgi_pass unix:/var/run/php/php8.4-fpm.sock;
        fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
        include fastcgi_params;
    }

    location / {
        try_files $uri $uri/ /index.html;
    }

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/www.hpsaviation.com/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/www.hpsaviation.com/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot
}
